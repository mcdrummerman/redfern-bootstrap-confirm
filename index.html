Example Usage:

    // initialize default confirm
    $(selector).confirm();
    // Show your new confirm
    $(selector).confirm('show');
    // Set body of current confirm box
    $(selector).confirm('setBody', 'new body!!!!!')
    // Set title of current confirm box
    $(selector).confirm('setTitle', 'new body!!!!!')
    // Hide cancel button of current confirm box
    $(selector).confirm('hideCancelButton')
    // Show cancel button of current confirm box
    $(selector).confirm('showCancelButton')
    // Re-initialize the confirm box with new options. Defaults for those unsupplied will be used 
    $(selector).confirm({title: 're-initialized title'});
    // Destroy current confirm box
     $(selector).confirm('destroy')
